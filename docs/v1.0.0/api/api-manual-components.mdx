This page demonstrates how to document your REST API endpoints using Specra's API components.

## Get User

<ApiEndpoint
  method="GET"
  path="/api/users/:id"
  summary="Retrieve a single user by ID"
>

### Path Parameters

<ApiParams
  title="Path Parameters"
  params={[
    {
      name: "id",
      type: "string",
      required: true,
      description: "The unique identifier of the user"
    }
  ]}
/>

### Query Parameters

<ApiParams
  title="Query Parameters"
  params={[
    {
      name: "fields",
      type: "string",
      description: "Comma-separated list of fields to include in the response",
      default: "all"
    },
    {
      name: "include_metadata",
      type: "boolean",
      description: "Whether to include user metadata",
      default: "false"
    }
  ]}
/>

### Responses

<ApiResponse
  status={200}
  description="Success"
  example={{
    "id": "user_123",
    "name": "John Doe",
    "email": "john@example.com",
    "created_at": "2024-01-15T10:30:00Z"
  }}
/>

<ApiResponse
  status={404}
  description="User not found"
  example={{
    "error": "User not found",
    "code": "USER_NOT_FOUND"
  }}
/>

### Try it out

<ApiPlayground
  method="GET"
  path="/api/users/user_123"
  baseUrl="https://api.example.com"
/>

</ApiEndpoint>

## Create User

<ApiEndpoint
  method="POST"
  path="/api/users"
  summary="Create a new user account"
>

### Request Body

<ApiParams
  title="Body Parameters"
  params={[
    {
      name: "name",
      type: "string",
      required: true,
      description: "Full name of the user"
    },
    {
      name: "email",
      type: "string",
      required: true,
      description: "Email address (must be unique)"
    },
    {
      name: "role",
      type: "string",
      description: "User role",
      default: "user"
    }
  ]}
/>

### Responses

<ApiResponse
  status={201}
  description="User created successfully"
  example={{
    "id": "user_124",
    "name": "Jane Smith",
    "email": "jane@example.com",
    "role": "user",
    "created_at": "2024-01-15T11:00:00Z"
  }}
/>

<ApiResponse
  status={400}
  description="Validation error"
  example={{
    "error": "Invalid request body",
    "details": {
      "email": "Email already exists"
    }
  }}
/>

### Try it out

<ApiPlayground
  method="POST"
  path="/api/users"
  baseUrl="https://api.example.com"
  defaultBody={JSON.stringify({
    name: "Jane Smith",
    email: "jane@example.com",
    role: "user"
  }, null, 2)}
/>

</ApiEndpoint>

## Authentication

All API requests require authentication using a Bearer token:

```bash
curl -H "Authorization: Bearer your_api_key" \
  https://api.example.com/api/users/user_123
```

Get your API key from the [developer dashboard](https://example.com/dashboard).
