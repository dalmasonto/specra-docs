Diese Seite demonstriert, wie Sie Ihre REST-API-Endpunkte mit den API-Komponenten von Specra dokumentieren.

## Benutzer abrufen

<ApiEndpoint
  method="GET"
  path="/api/users/:id"
  summary="Einen einzelnen Benutzer per ID abrufen"
>

### Pfadparameter

<ApiParams
  title="Pfadparameter"
  params={[
    {
      name: "id",
      type: "string",
      required: true,
      description: "Die eindeutige Kennung des Benutzers"
    }
  ]}
/>

### Query-Parameter

<ApiParams
  title="Query-Parameter"
  params={[
    {
      name: "fields",
      type: "string",
      description: "Kommagetrennte Liste der Felder, die in die Antwort aufgenommen werden sollen",
      default: "all"
    },
    {
      name: "include_metadata",
      type: "boolean",
      description: "Ob Benutzermetadaten enthalten sein sollen",
      default: "false"
    }
  ]}
/>

### Antworten

<ApiResponse
  status={200}
  description="Erfolg"
  example={{
    "id": "user_123",
    "name": "John Doe",
    "email": "john@example.com",
    "created_at": "2024-01-15T10:30:00Z"
  }}
/>

<ApiResponse
  status={404}
  description="Benutzer nicht gefunden"
  example={{
    "error": "User not found",
    "code": "USER_NOT_FOUND"
  }}
/>

### Ausprobieren

<ApiPlayground
  method="GET"
  path="/api/users/user_123"
  baseUrl="https://api.example.com"
/>

</ApiEndpoint>

## Benutzer erstellen

<ApiEndpoint
  method="POST"
  path="/api/users"
  summary="Ein neues Benutzerkonto erstellen"
>

### Request Body

<ApiParams
  title="Body-Parameter"
  params={[
    {
      name: "name",
      type: "string",
      required: true,
      description: "Vollständiger Name des Benutzers"
    },
    {
      name: "email",
      type: "string",
      required: true,
      description: "E-Mail-Adresse (muss eindeutig sein)"
    },
    {
      name: "role",
      type: "string",
      description: "Benutzerrolle",
      default: "user"
    }
  ]}
/>

### Antworten

<ApiResponse
  status={201}
  description="Benutzer erfolgreich erstellt"
  example={{
    "id": "user_124",
    "name": "Jane Smith",
    "email": "jane@example.com",
    "role": "user",
    "created_at": "2024-01-15T11:00:00Z"
  }}
/>

<ApiResponse
  status={400}
  description="Validierungsfehler"
  example={{
    "error": "Invalid request body",
    "details": {
      "email": "Email already exists"
    }
  }}
/>

### Ausprobieren

<ApiPlayground
  method="POST"
  path="/api/users"
  baseUrl="https://api.example.com"
  defaultBody={JSON.stringify({
    name: "Jane Smith",
    email: "jane@example.com",
    role: "user"
  }, null, 2)}
/>

</ApiEndpoint>

## Authentifizierung

Alle API-Anfragen erfordern eine Authentifizierung mit einem Bearer-Token:

```bash
curl -H "Authorization: Bearer your_api_key" \
  https://api.example.com/api/users/user_123
```

Erhalten Sie Ihren API-Schlüssel vom [Entwickler-Dashboard](https://example.com/dashboard).
